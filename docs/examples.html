WebSocket Events Lib - Examples

Listen to a event
use FoxWorn3365\WebSocketEvents\Core;

require 'include_all.php';

$core = new Core("localhost", 1991, "jhwebf384Bhsv98HJBv378jhbsdAjd");

$core->on('player_jump', function(Core $core, object $data) {
    // $data->player is a Part\Player object
    echo "\nPlayer {$data->player->name} jumped!";
});

$core->on('ready', function(Core $core) {
    echo "READY";
});

$core->connect();
